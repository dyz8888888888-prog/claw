**`core/kpl_client.py` 核心逻辑：**

```python
class KPLClient:
    """kaipanla-crawler 封装 (MIT协议, 免费)"""
    
    def __init__(self):
        from kaipanla_crawler import KaipanlaCrawler
        self._crawler = KaipanlaCrawler()
        self._cache = {}  # {date: data}
    
    def get_daily_summary(self, date=None) -> dict:
        """获取单日市场概览: 涨跌停数/新高数/上涨比例等"""
        return self._crawler.get_daily_data(date)
    
    def get_limit_up_sectors(self, date=None) -> dict:
        """获取涨停原因板块: 每个概念下的涨停股含涨停时间/封单/原因/标签"""
        return self._crawler.get_limit_up_sectors(date)
    
    def get_sector_ranking(self, date=None):
        """获取板块排行 (含资金流)"""
        return self._crawler.get_sector_ranking(date)
    
    def get_sentiment(self):
        """获取多空风向标 + 情绪指标"""
        return {
            'sentiment': self._crawler.get_sentiment_indicator(),
            'new_high': self._crawler.get_new_high_data(),
        }
```

**数据量级：** 每个交易日收盘后 (`15:30`) 跑一次即可，全量约 50KB JSON。

**`core/dragon_ranker.py` 核心逻辑：**

```python
class DragonRanker:
    """概念板块龙头排序引擎"""
    
    # 最终权重 (经过网格搜索优化)
    TIME_WEIGHT = 0.35    # 涨停时间分
    FENGDAN_WEIGHT = 0.35 # 封单力度分
    LIANBAN_WEIGHT = 0.30 # 连板高度分
    
    def rank_sector(self, sector: dict) -> list[dict]:
        """
        输入: kaipanla-crawler 返回的一个 sector
        输出: 排序后的 dragos = [
          {rank: 1, name, code, time_score, fengdan_score, lianban_score, total_score, limit_up_time, 概念标签, 涨停原因}
        ]
        """
        stocks = sector.get('stocks', [])
        for s in stocks:
            s['time_score'] = self._calc_time_score(s.get('涨停时间', ''))  # 10:00前=100, 渐降
            s['fengdan_score'] = self._calc_fengdan_score(
                float(s.get('封单额', 0) or 0),
                float(s.get('流通市值', 0) or 1)  # 封单/流通市值
            )
            s['lianban_score'] = int(s.get('连板天数', 0) or 0) * 10
            s['total_score'] = (
                s['time_score'] * self.TIME_WEIGHT +
                s['fengdan_score'] * self.FENGDAN_WEIGHT +
                s['lianban_score'] * self.LIANBAN_WEIGHT
            )
        
        stocks.sort(key=lambda x: -x['total_score'])
        for i, s in enumerate(stocks):
            s['rank'] = i + 1
        return stocks
    
    def _calc_time_score(self, time_str: str) -> float:
        """涨停越早分越高: 9:30=100, 10:00=90, 10:30=80, 11:00=70, 13:30=60, 14:00=50, 14:30=40, 尾盘=30"""
        # 解析时间字符串, 线性衰减
        ...
    
    def _calc_fengdan_score(self, fengdan: float, mcap: float) -> float:
        """封单力度: 封单/流通市值 比例越大越好, 上限100"""
        ratio = fengdan / max(mcap, 1) * 100
        return min(ratio * 12.5, 100)  # 8%=100分
```

```python
class SentimentEngine:
    """六维市场情绪引擎 (参考知乎《超短复盘必备情绪指标》)"""
    
    # 六维度阈值表
    THRESHOLDS = {
        '过热': {'limit_up': 100, 'blast_ratio': 0.20, 'promotion': 0.60, 'up_down_ratio': 3.0, 'advice': '减仓防分歧'},
        '活跃': {'limit_up': 50,  'blast_ratio': 0.30, 'promotion': 0.40, 'up_down_ratio': 2.0, 'advice': '积极操作'},
        '温和': {'limit_up': 30,  'blast_ratio': 0.40, 'promotion': 0.25, 'up_down_ratio': 1.0, 'advice': '谨慎参与'},
        '退潮': {'limit_up': 15,  'blast_ratio': 0.50, 'promotion': 0.15, 'up_down_ratio': 0.5, 'advice': '控制仓位'},
        '冰点': {'limit_up': 0,   'blast_ratio': 0.99, 'promotion': 0.0,  'up_down_ratio': 0.0, 'advice': '空仓等待'},
    }
    
    def evaluate(self, kpl_summary: dict, kpl_limit_up: dict) -> dict:
        """
        返回:
        {
          phase: '活跃',          # 情绪阶段
          advice: '积极操作',      # 操作建议
          indicators: {
            limit_up_count: 68,   # 涨停家数
            blast_ratio: 0.25,    # 炸板率
            promotion_rate: 0.45, # 连板晋级率
            up_down_ratio: 2.3,   # 涨跌比
            new_high_count: 12,   # 百日新高数
            big_face_count: 2,    # 大面股数 (浮亏>8%)
            yesterday_limit_up_avg: 1.2,  # 昨日涨停均收益%
          },
          signals: ['涨停家数增多', '炸板率健康'],
          timestamp: '2026-06-23 09:45'  # kpl数据是收盘后, 盘中用快照实时
        }
        """
```

**关键设计决策：**

1. **盘中 vs 盘后**: kaipanla-crawler 数据收盘后才生成。盘中指标用以下替代：
   - `涨停数` → TDX 快照中 `stock_change_pct >= 9.5` 的数量
   - `炸板率` → 开盘啦API实时或 同花顺涨停池实时
   - `涨跌比` → 转债池内实时计算
   - `炸板/晋级` → **盘中无法从免费源获取**, 用盘后kpl数据做趋势参考

2. **数据源分配**：
   - 盘中 09:30-15:00 → TDX快照 + 转债池涨跌比 (轻量免费)
   - 盘后 15:30 → kaipanla-crawler 全量 (炸板/晋级/封单完整)
   - 次日开盘前 → 自动加载盘后计算结果作为当天基准
